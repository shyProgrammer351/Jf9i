문제

돈을 입력 받아, 500원, 100원, 50원, 10원의 개수를 계산하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
잔돈을 입력하세요: 3210
3210원을 500원짜리 동전 6개, 100원짜리 동전 2개, 50원짜리 동전 0개, 10원짜리 동전 1개로 거슬러 드립니다.

프로그램 코드

	#include <stdio.h>
	#include <stdlib.h>
	
	typedef struct _Coin { 					
		int _500;
		int _100;
		int _50;
		int _10;
	} Coin;    
	
1	int getPayment()
	{
		int payment = 0;
	
		printf( "잔돈을 입력하세요: " );
		scanf( "%d", &payment ); 						
	
2		return payment;
	}
	
3	Coin* getCoins( int payment )			
	{
4		Coin* coin = 						
5			     (Coin*)malloc( sizeof(Coin) );		
	
		coin->_500 = payment / 500;							
		payment = payment % 500;						
	
		coin->_100 = payment / 100;							
		payment = payment % 100;					
	
		coin->_50  = payment / 50;							
		payment = payment % 50;						
	
		coin->_10  = payment / 10;							
		payment = payment % 10;					
	
6		return coin;
	}
	
7	int main()
	{
		int payment = 0;						
		Coin* coin = NULL;								
	
8		payment =
9			  getPayment();
	
10		coin = 
11		        getCoins( payment );
	
12		printf( "%d원을 500원짜리동전 %d개,100원짜리동전 %d개,50원짜리동전 %d개,10원짜리동전 %d개로 거슬러 드립니다.", 
			payment, coin->_500, coin->_100, coin->_50, coin->_10 );		
	
13		free( coin );									
	
14		return 0;
	}
