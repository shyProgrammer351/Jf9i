문제

미로 맵을 출력하는 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요 
[참고] 하늘색(바탕) \033[44m, 하늘색(글자) \033[34m, 빨강색(글자) \033[31m, 색상초기화 \033[0m
                  
  옷            문

코드 빈칸 채우기

public class MazeMap 
{ 
	public static void main( String[] args ) { 
		char map = { 
		                   { 'W''W''W''W''W''W''W''W''W' }, 
		                   { 'W''A'' '' ''W'' '' '' ''E' }, 
		                   { 'W''W''W'' ''W'' ''W''W''W' }, 
		                   { 'W'' '' '' ''W'' '' '' ''W' }, 
		                   { 'W'' ''W''W''W''W''W'' ''W' }, 
		                   { 'W'' '' '' '' '' '' '' ''W' }, 
		                   { 'W''W''W''W''W''W''W''W''W' }  
		                 } ; 
				     	 
		( int row = ; row  map.; row ) { 
			( int column = ; column  map.; column) { 
				( maprowcolumn  'W' ) 
					System.out.print( " \033[44m   \033[0m" ); 
				( maprowcolumn  ' ' ) 
					System.out.print( "  " ); 
				( maprowcolumn  'E' ) 
					System.out.print( " \033[34m문 \033[0m" ); 
				( maprowcolumn  'A' ) 
					System.out.print( " \033[31m옷 \033[0m" ); 
			}	 
			System.out.println(); 
		}	 
		 
	} 
}
