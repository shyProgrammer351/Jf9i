문제

영어 단어를 입력하면 영한 사전을 열고 그 뜻을 찾아서 출력하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
--------- 파일 dictionary.txt의 내용 --------- 
apple의 뜻은 사과입니다.
banana의 뜻은 바나나입니다.
carrot의 뜻은 당근입니다.
...

영어단어를 입력하세요 : apple
apple의 뜻은 사과입니다.

프로그램 코드

	#include <stdio.h>
	#include <string.h>
	
	#define MAX_STR_LEN 1024	

1	int main()
	{
		char word[MAX_STR_LEN] = "";
		char meaning[MAX_STR_LEN] = "";
2		FILE *dictionary = NULL;						
	
		printf( "영어단어를 입력하세요 : " );
3		scanf( "%s", word );

4		if ( ( dictionary = fopen( "dictionary.txt", "rt" ) ) == NULL )		
		{
			printf( "[오류] 파일을 열 수 없습니다!" );
5			return -1;
		}
	
6		while ( fgets( meaning, MAX_STR_LEN, dictionary ) != NULL ) 		
		{
7			if ( !strncmp( word,  meaning, strlen( word ) ) )
			{
				printf( "%s", meaning );
8				break;
			}
		}

9		fclose( dictionary );							
	
10		return 0;
	}
