프로그램시작 
   파일 열기 
   원본 파일 끝까지 한 줄씩 읽기 
   복사본 파일에 한 줄씩 쓰기 
   파일 닫기 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter09/FileCopy.c 

	#include <stdio.h> 

	 

	// 문자열 최대 길이는 1024 

	#define MAX_STR_LEN 1024	 

 

	// 프로그램시작 

1	int main() 

	{ 

		char line[MAX_STR_LEN] = ""; 

	 

		// 파일 열기 

		FILE *inFile = fopen( "원본.txt", "rt" );			 

2		FILE *outFile = fopen( "복사본.txt", "wt" );			 

3		if ( ( inFile == NULL ) || ( outFile == NULL ) )		 

		{ 

			printf( "[오류] 파일을 열 수 없습니다!" ); 

4			return -1; 

		} 

 

		// 원본 파일 끝까지 한 줄씩 읽기 

5		while ( fgets( line, MAX_STR_LEN, inFile ) != NULL ) 	 

		{ 

			// 복사본 파일에 한 줄씩 쓰기 

6			fprintf( outFile, "%s", line ); 			 

		} 

 

		// 파일 닫기 

		fclose( inFile );						 

7		fclose( outFile );							 

	 

		// 프로그램 종료 

8		return 0; 

	}
