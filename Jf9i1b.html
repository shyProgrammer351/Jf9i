package network; 
import java.net.; 
import java.util.ArrayList; 
 
public class MultiClientServer extends Network 
{ 
	private ArrayList clients; 
 
	public MultiClientServer() 
	{  
		super(); 
		clients = new ArrayList(); 
	} 
 
	public void connectAsMultiClientServer() 
	{ 
		try  
		{ 
			serverSocket = new ( serverPort );  
			System.out.println( "[Server] 서버 대기중" ); 
 
			while ( true ) 
			{ 
				socket =  
						serverSocket.();   
				Client client =  
							new Client( this ); 
				System.out.println( "[Server] 클라이언트 연결 : " + client ); 
				clients.add( client );  
				client.();   
			}
		} 
		catch ( Exception e ) 
		{  
			e.printStackTrace(); 
		}  
	} 
	 
	public void ( String data )  
	{ 
		for( Client client : clients ) 
		{ 
			client.write( data );	 
		}  
	} 
 
	public void disconnect()  
	{ 
		for( Client client : clients ) 
		{ 
			client.();  
			client.disconnect();  
			clients.remove( client ); 
		} 
		 
		super.disconnect(); 
	} 
}

	private PrintWriter out;

	{ 
		try  
		{ 
			this.server = server;

			this.out = new PrintWriter( server.socket.getOutputStream(), true ); 
		}  
		catch ( Exception e ) 
		{  
			e.printStackTrace(); 
		}  
	}

				server.( data );   
			} 
		}  
		catch ( Exception e ) 
		{  
			e.printStackTrace(); 
		}  
	}

		try  
		{

			}

			{

package network; 
import java.io.BufferedReader; 
import java.io.InputStreamReader; 
import java.io.PrintWriter; 
import java.net.; 
import java.net.; 
 
public class Network  
{ 
	protected int serverPort; 
	protected  serverSocket;  
	protected  socket;  
	 
	private BufferedReader in; 
	private PrintWriter out;  
	private  waitForCounterpart;  
 
	public Network() 
	{	 
		serverPort = 7700; 
		serverSocket = null; 
		socket = null; 
		in = null; 
		out = null; 
		waitForCounterpart = null; 
	} 
	 
	public void connectAsServer(  obj )   
	{ 
		try  
		{ 
			serverSocket = new ( serverPort );  
			socket =  
			         serverSocket.();    
			in = new BufferedReader( new InputStreamReader( socket.getInputStream() ) ); 
			out = new PrintWriter( socket.getOutputStream(), true ); 
			waitForCounterpart = new ( obj );  
			waitForCounterpart.();   
		} 
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	}  
 
	public void connectAsClient( String serverIP,  obj )   
	{ 
		try  
		{ 
			socket =  
			         new ( serverIP, serverPort );  
			in = new BufferedReader( new InputStreamReader( socket.getInputStream() )); 
			out = new PrintWriter( socket.getOutputStream(), true ); 
			waitForCounterpart = new ( obj );  
			waitForCounterpart.();	 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	} 
	 
	public String read()  
	{ 
		try  
		{ 
			if ( this.isConnecting() == true ) 
				return  
				       in.readLine();
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
		 
		return null; 
	} 
	 
	public void write( String data )  
	{ 
		try  
		{ 
			if ( this.isConnecting() == true ) 
				out.println( data ); 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		}	 
	} 
	 
	public void disconnect()  
	{ 
		try  
		{ 
			if( waitForCounterpart != null ) 
			{ 
				waitForCounterpart.();	   
				waitForCounterpart = null; 
			} 
			if( in != null ) 
			{ 
				in.close(); 
				in=null; 
			}			 
			if( out != null ) 
			{ 
				out.flush(); 
				out.close(); 
				out = null; 
			} 
			if( socket != null ) 
			{ 
				socket.close(); 
				socket = null; 
			} 
			if( serverSocket != null ) 
			{ 
				serverSocket.close(); 
				serverSocket = null; 
			} 
		}  
		catch ( Exception e ) 
		{ 
			e.printStackTrace(); 
		} 
	} 
 
	public boolean isConnecting()  
	{ 
		if ( ( socket != null )  
				&& ( in != null )  
				&& ( out != null )  
				&& ( waitForCounterpart != null )  
				&& ( waitForCounterpart.getState() != .State.TERMINATED ) )    
			return true; 
		else 
			return false; 
	} 
}
