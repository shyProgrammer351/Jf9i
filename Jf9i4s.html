미로 맵을 출력하는 문제입니다 실행순서를 클릭하세요 
[참고] 하늘색(바탕) \033[44m, 하늘색(글자) \033[34m, 빨강색(글자) \033[31m, 색상초기화 \033[0m
                  
  옷            문

프로그램 코드

		

	public class MazeMap

	{

1		public static void main( String[] args ) {

2			char[][] map = {

			                   { 'W','W','W','W','W','W','W','W','W' },

			                   { 'W','W','W',' ','W',' ','W','W','W' },

			                   { 'W',' ',' ',' ','W',' ',' ',' ','W' },

			                   { 'W',' ','W','W','W','W','W',' ','W' },

			                   { 'W',' ',' ',' ',' ',' ',' ',' ','W' },

			                   { 'W','W','W','W','W','W','W','W','W' }

			                 } ;

					     	

3			for( int row = 0; row < map.length; row++ ) {

4				for( int column = 0; column < map[0].length; column ++ ) {

5					if( map[row][column] == 'W' )

6						System.out.print( " \033[44m   \033[0m" );

7					else if( map[row][column] == ' ' )

8						System.out.print( "  " );

9					else if( map[row][column] == 'E' )

12						System.out.print( " \033[31m옷 \033[0m" );

				}

13				System.out.println();

			}

			

14		}

	}
