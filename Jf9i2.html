알고리즘

프로그램 시작 
   점수 입력 
   점수 정렬 
      마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 
   정렬된 점수 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter09/ScoreSort.java

	import java.util.Scanner;

		 

	public class ScoreSort

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			Scanner scan = new Scanner( System.in );

2			int[] scores = { 0, 0, 0 };

			 

			// 점수 입력  

3			for ( int index = 0; index < scores.length; index++ ) {

4				System.out.print( "0점 ~ 100점 사이의 점수를 입력하세요: " );

5				scores[index] = scan.nextInt();

			}

			 

			// 점수 정렬 

			// 마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 

6			for ( int last = scores.length - 1; 0 <= last; last-- ) {

				// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

7				for ( int index = 0; index < last; index++ ) {

					// 앞이 뒤보다 크면 두 값을 맞바꾸기하면서 뒤쪽에 큰 값을 저장 

8					if ( scores[index] > scores[index+1] ) {

9						int temp = scores[index];

10						scores[index] = scores[index+1];

11						scores[index+1] = temp;

					}

				}

			}

		 

			// 정렬된 점수 출력 

12			for ( int index = 0; index < scores.length; index++ )

13				System.out.print( scores[index] + " " );

		 

		// 프로그램 종료 

14		}

	}
