배열   동일한 자료형의 여러 변수를 하나로 묶어서 표현하며, 많은 데이터를 동일하게 처리하는 경우 편리

0   배열의 여러 칸 중에서 첫번째 칸을 나타내는 번호

length   배열 크기
배열의 인덱스번호는 0부터 시작하므로, 배열크기 = 인덱스끝번호 + 1 = ( column - 1 ) + 1 = column

for( 자료형 변수명 : 배열명 ) { 명령문 }  첫 배열요소부터 끝 배열요소까지 하나씩 변수에 저장 후 명령문 반복

import java.util.Scanner;

public class Main {
  public static void main( String[] args ) {
    Scanner scan = new Scanner( System.in );

    System.out.print("1~12 사이의 월을 입력하세요: ");
    int month = scan.nextInt();

    String[] monthName = { "January", "February",
            "March", "April", "May", "June", "July", "August",
            "September", "October", "November", "December" };
    System.out.print( month + "월은 "
                            + monthName[ month - 1 ] + "입니다" );
  }
}
































import java.util.Scanner;

public class Main {
  public static void main( String[] args ) {
    Scanner scan = new Scanner( System.in );

    System.out.print("1~12 사이의 월을 입력하세요: ");
    int month = scan.nextInt();
   
    switch ( month ) {
      case 1:
        System.out.println( "1월은 January입니다" );
        break;
      case 2:
        System.out.println( "2월은 February입니다" );
        break;
      case 3:
        System.out.println( "3월은 March입니다" );
        break;
      case 4:
        System.out.println( "4월은 April입니다" );
        break;
      case 5:
        System.out.println( "5월은 May입니다" );
        break;
      case 6:
        System.out.println( "6월은 June입니다" );
        break;
      case 7:
        System.out.println( "7월은 July입니다" );
        break;
      case 8:
        System.out.println( "8월은 August입니다" );
        break;
      case 9:
        System.out.println( "9월은 September입니다" );
        break;
      case 10:
        System.out.println( "10월은 October입니다" );
        break;
      case 11:
        System.out.println( "11월은 November입니다" );
        break;
      default:
        System.out.println( "12월은 December입니다" );
    }
  }
}
