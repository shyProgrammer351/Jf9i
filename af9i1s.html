문제

파일 원본을 읽어 파일 복사본을 생성하는 파일 복사 문제입니다 실행순서를 클릭하세요 
--------- 파일 원본.txt의 내용 --------- 
과연 파일 복사는 잘 될까요?
파일 복사가 잘 되면 좋겠어요^^

프로그램 코드

	#include <stdio.h>


	#define MAX_STR_LEN 1024	


1	int main()

	{

		char line[MAX_STR_LEN] = "";


		FILE *inFile = fopen( "원본.txt", "rt" );			

2		FILE *outFile = fopen( "복사본.txt", "wt" );			

3		if ( ( inFile == NULL ) || ( outFile == NULL ) )		

		{

			printf( "[오류] 파일을 열 수 없습니다!" );

4			return -1;

		}


5		while ( fgets( line, MAX_STR_LEN, inFile ) != NULL ) 	

		{

6			fprintf( outFile, "%s", line ); 			

		}


		fclose( inFile );						

7		fclose( outFile );							


8		return 0;

	}
